const Find_Digits = (N) =>
{
var count = 0;
if (N >= 1) ++count;
while (N / 10 >= 1) {
N /= 10;
++count;
}
return count;
};
const Number_Sum = (N) =>
{
let sum = 0;
while(N){
sum = sum + ( N % 10)
N = parseInt( N / 10)
}
return sum;
};
